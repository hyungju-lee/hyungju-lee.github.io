## Level 1. 문제 1

```javascript
// 입력
["sun", "bed", "car"], 1
["abce", "abcd", "cdx"], 2

// 출력
["car", "bed", "sun"]
["abcd", "abce", "cdx"]
```

```javascript
// 내가 푼 답
const solution = (arr, n) => {
    return arr.sort((a, b) => {
        return a[n] < b[n] ? -1 : a[n] > b[n] ? 1 : a < b ? -1 : a > b ? 1: 0;
    })
}
```
