## Level 1. 문제 6

```javascript
// 입력
solution(["a", "b", "z", "a", "e"], ["a", "b"])

// 출력
["z", "a", "e"]
```

```javascript
// 내가 푼 답
const solution = (participant, completion) => {
    let arr = [];
    participant.map(a => {
        let b = false;
        for (let i=completion.length-1; i>=0; i--) {
            if(a===completion[i]) {
                b = true;
                completion.splice(i, 1);
                continue;
            }
        }
        if(!b) {
            arr.push(a);
        }
    })
    return arr;
}
```
